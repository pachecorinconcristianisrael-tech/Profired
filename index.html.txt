  <script>
    // Verificar sesión
    const usuario = JSON.parse(localStorage.getItem("usuarioActivo"));
    if (usuario && usuario.correo) {
      document.getElementById("bienvenida").textContent = `Bienvenido, ${usuario.correo}`;
    } else {
      window.location.href = "login.html";
    }

    function cerrarSesion() {
      localStorage.removeItem("usuarioActivo");
      window.location.href = "login.html";
    }

    // Tabs (corregido con event)
    function mostrarTab(event, tab) {
      document.querySelectorAll('.tab-content').forEach(div => div.classList.remove('active'));
      document.getElementById(tab).classList.add('active');
      document.querySelectorAll('nav button').forEach(btn => btn.classList.remove('active'));
      event.target.classList.add('active');
    }

    // Carrito con suma de precios y eliminación
    let total = 0;
    function agregarCarrito(nombre, precio) {
      const lista = document.getElementById("listaCarrito");
      const li = document.createElement("li");
      li.innerHTML = `
        <span>${nombre}</span>
        <span>$${precio}</span>
        <button onclick="quitarDelCarrito(this, ${precio})">Quitar</button>
      `;
      lista.appendChild(li);
      total += precio;
      actualizarTotal();
    }

    function quitarDelCarrito(button, precio) {
      const li = button.parentElement;
      li.remove();
      total -= precio;
      if (total < 0) total = 0;
      actualizarTotal();
    }

    function actualizarTotal() {
      document.getElementById("total").textContent = `Total: $${total}`;
    }

    function comprar() {
      const lista = document.getElementById("listaCarrito");
      if (!lista.children.length) {
        alert("Tu camión está vacío. Añade profesionistas primero.");
        return;
      }
      alert(`? Compra registrada.\nTotal: $${total}\nNos pondremos en contacto para coordinar los servicios.`);
      // Limpia el carrito
      lista.innerHTML = "";
      total = 0;
      actualizarTotal();
    }

    // Postular (inserta perfil con precio y botón de carrito)
    function postular() {
      const nombre = document.getElementById("nombre").value.trim();
      const edad = document.getElementById("edad").value.trim();
      const genero = document.getElementById("genero").value;
      const profesion = document.getElementById("profesion").value.trim();
      const precio = parseInt(document.getElementById("precio").value, 10);
      const ubicacion = document.getElementById("ubicacion").value.trim();

      if (!nombre || !edad || !profesion || isNaN(precio)) {
        alert("Por favor, completa todos los campos obligatorios.");
        return;
      }

      const empleadosTab = document.getElementById("empleados");
      const nuevoPerfil = document.createElement("div");
      nuevoPerfil.classList.add("perfil");
      nuevoPerfil.innerHTML = `
        <img src="foto_default.jpg" alt="Foto perfil">
        <h3>${nombre}</h3>
        <p>Edad: ${edad}${genero ? ' • ' + genero : ''}</p>
        <p>Profesión: ${profesion}</p>
        ${ubicacion ? `<p>Ubicación: ${ubicacion}</p>` : ''}
        <p>Precio por hora: $${precio}</p>
        <button onclick="agregarCarrito('${nombre} - ${profesion}', ${precio})">Añadir al carrito</button>
      `;
      // Inserta antes del carrito
      empleadosTab.insertBefore(nuevoPerfil, document.getElementById("carrito"));

      // Limpia el formulario
      document.getElementById("nombre").value = "";
      document.getElementById("edad").value = "";
      document.getElementById("genero").value = "";
      document.getElementById("profesion").value = "";
      document.getElementById("precio").value = "";
      document.getElementById("ubicacion").value = "";

      alert("? Te has postulado exitosamente. Ahora apareces en la lista de empleados.");
      // Cambia a la pestaña Empleados
      const empleadosBtn = Array.from(document.querySelectorAll('nav button')).find(b => b.textContent.includes('Empleados'));
      mostrarTab({ target: empleadosBtn }, 'empleados');
    }
